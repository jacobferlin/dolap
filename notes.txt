cube(ProductCube) %>%
  measure(MerchandiseSalesQty) %>%
  group_by(Date.Year, Product.ProductGroupCurrent) %>%
  filter(Product.ProductBusinessGroup == "Jotex") %>%
  order(MerchandiseSalesQty, desc) %>%
  top(10)

SELECT 
  [Measures].[MerchandiseSalesQty] ON 0, 
  HEAD ( 
    ORDER( 
      ( [Date].[Year].CHILDREN, 
        [Product].[ProductGroupCurrent].CHILDREN ), 
      [Measures].[MerchandiseSalesQty], 
      BDESC),
    10 ) ON 1 
FROM [ProductCube]
